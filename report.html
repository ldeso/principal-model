I_T histogram vs. Dufresne closed-form moments

itHist = {
  const { edges, counts } = run.itSampleHistogram;
  const total = counts.reduce((a,b) => a+b, 0);
  return counts.map((c,i) => ({
    x0: edges[i], x1: edges[i+1], density: c / (total * (edges[i+1]-edges[i]))
  }));
}
Plot.plot({
  width: 720,
  height: 300,
  x: { label: "I_T" },
  y: { label: "empirical density" },
  marks: [
    Plot.rectY(itHist, { x1: "x0", x2: "x1", y: "density", fill: "#4c78a8", fillOpacity: 0.7 }),
    Plot.ruleX([run.closed.IT.mean], { stroke: "black" }),
    Plot.ruleX([run.closed.IT.mean - Math.sqrt(run.closed.IT.variance),
                run.closed.IT.mean + Math.sqrt(run.closed.IT.variance)],
               { stroke: "black", strokeDasharray: "3 3" })
  ]
})

Solid rule: E[I_T] (Dufresne). Dashed rules: E[I_T] ± SD.

Break-even quote surface Q*(μ, T)

qstar = FileAttachment("data/qstar-surface.json").json()
qstarRecords = qstar.mus.flatMap((mu, i) =>
  qstar.Ts.map((T, j) => ({ mu, T, Q: qstar.values[i][j] }))
)
Plot.plot({
  width: 720,
  height: 360,
  marginLeft: 60,
  x: { label: "T (years)" },
  y: { label: "Q* / ((1+f)·P·S_0)" },
  color: { legend: true, label: "μ" },
  marks: [
    Plot.ruleY([1], { stroke: "#888", strokeDasharray: "2 4" }),
    Plot.line(qstarRecords, {
      x: "T",
      y: d => d.Q / ((1 + run.params.f) * run.params.P * run.params.S0),
      stroke: "mu",
      strokeWidth: 2
    }),
    Plot.dot(qstarRecords, {
      x: "T",
      y: d => d.Q / ((1 + run.params.f) * run.params.P * run.params.S0),
      fill: "mu"
    })
  ]
})

The μ = 0 curve collapses onto the dashed unit line — at zero drift, Q* = (1+f)·P·S_0, i.e., the fee-based time-zero quote.
